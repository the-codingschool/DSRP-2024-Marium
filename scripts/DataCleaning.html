data_sml <- data_clean %>%
  select(ride_id, started_at, ended_at) %>%
  separate(started_at, into = c("start_date", "start_time"), sep = " ") %>%
  separate(ended_at, into = c("end_date", "end_time"), sep = " ") %>%
  mutate(started_at = start_date,
         ended_at = end_date) %>%
